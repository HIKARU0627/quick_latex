\begin{frame}{付録：アルゴリズムの詳細}
    \begin{algorithm}[H]
        \caption{提案アルゴリズム（詳細版）}
        \KwIn{データセット $D$, パラメータ $\theta$}
        \KwOut{予測結果 $Y$}
        \BlankLine
        $Y \leftarrow \emptyset$\;
        \For{$x \in D$}{
            $x' \leftarrow \text{preprocess}(x)$\;
            $f \leftarrow \text{extract\_features}(x')$\;
            $y \leftarrow \text{predict}(f, \theta)$\;
            $Y \leftarrow Y \cup \{y\}$\;
        }
        \Return{$Y$}\;
    \end{algorithm}
\end{frame}
